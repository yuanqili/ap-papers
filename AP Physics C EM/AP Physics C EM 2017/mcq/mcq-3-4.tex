\begin{questions}\setcounter{question}{2}\question

\begin{oneparchoices}
\choice $4 / 1$
\choice $2 / 1$
\choice $1 / 1$
\choice $1 / 2$
\choice $1 / 4$
\end{oneparchoices}\end{questions}
